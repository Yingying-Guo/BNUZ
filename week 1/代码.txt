	XSS Test Part3:<br><input type="text" id='text3'>
	<button onclick="search3()">Search</button><br>
	<script>
		function search3()
		{
			var text=document.getElementById('text3').value;
			var newText=text.replace('&','&amp;');
			newText = newText.replace('<','&lt;');
			newText = newText.replace('>','&gt;');
			newText = newText.replace('"','&quot;');
			newText = newText.replace(' ','&nbsp;');
			document.write(newText);
		}
	</script>

	XSS Test Part4:<br><input type="text" id='text4'>
	<button onclick="search4()">Search</button><br>
	<script>
		function search4()
		{
			var reg = /<[\s\S]*s[\s\S]*c[\s\S]*r[\s\S]*i[\s\S]*p[\s\S]*t[\s\S]script>/i;
			var text=document.getElementById('text4').value;
			var newText=text.replace(reg,'');
			document.write(newText);
		}
	</script>
